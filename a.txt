#include <stdio.h> #include <stdlib.h> #include <cuda_runtime.h> #define STB_IMAGE_IMPLEMENTATION #include "stb_image.h" #define STB_IMAGE_WRITE_IMPLEMENTATION #include "stb_image_write.h" // ================= READ JPG ================= unsigned char* read_jpg(const char* filename, int* width, int* height) { int channels; unsigned char* data = stbi_load(filename, width, height, &channels, 1); if (!data) { printf("Error loading image: %s\n", stbi_failure_reason()); exit(1); } return data; } // ================= SUBTRACTION ================= __global__ void subtract(unsigned char* prev, unsigned char* curr, unsigned char* out, int n) { int i = blockIdx.x * blockDim.x + threadIdx.x; if (i < n) { int diff = curr[i] - prev[i]; if (diff < 0) diff = -diff; out[i] = diff; } } // ================= THRESHOLD ================= __global__ void threshold(unsigned char* in, unsigned char* out, int n, int T) { int i = blockIdx.x * blockDim.x + threadIdx.x; if (i < n) { out[i] = (in[i] > T) ? 255 : 0; } } // ================= MORPHOLOGY ================= __global__ void erosion(unsigned char* in, unsigned char* out, int width, int height) { int x = blockIdx.x * blockDim.x + threadIdx.x; int y = blockIdx.y * blockDim.y + threadIdx.y; if (x >= width || y >= height) return; if (x == 0 || y == 0 || x == width-1 || y == height-1) { out[y * width + x] = in[y * width + x]; return; } int min = 255; for (int dx = -1; dx <= 1; dx++) { for (int dy = -1; dy <= 1; dy++) { int val = in[(y+dy)*width + (x+dx)]; if (val < min) min = val; } } out[y * width + x] = min; } // ================= MAIN ================= int main() { int width, height; // CUDA timing cudaEvent_t start, stop; cudaEventCreate(&start); cudaEventCreate(&stop); for (int i = 2; i <= 50; i++) { char prev_name[100], curr_name[100], out_name[100]; sprintf(prev_name, "dataset/input/in%06d.jpg", i-1); sprintf(curr_name, "dataset/input/in%06d.jpg", i); sprintf(out_name, "dataset/output/out%06d.jpg", i); unsigned char* h_prev = read_jpg(prev_name, &width, &height); unsigned char* h_curr = read_jpg(curr_name, &width, &height); int size = width * height; unsigned char *d_prev, *d_curr, *d_diff, *d_thresh, *d_morph; cudaMalloc(&d_prev, size); cudaMalloc(&d_curr, size); cudaMalloc(&d_diff, size); cudaMalloc(&d_thresh, size); cudaMalloc(&d_morph, size); cudaMemcpy(d_prev, h_prev, size, cudaMemcpyHostToDevice); cudaMemcpy(d_curr, h_curr, size, cudaMemcpyHostToDevice); int threads = 256; int blocks = (size + threads - 1) / threads; dim3 t2(16,16); dim3 b2((width+15)/16, (height+15)/16); // START TIMER cudaEventRecord(start); // -------- PIPELINE -------- subtract<<<blocks, threads>>>(d_prev, d_curr, d_diff, size); threshold<<<blocks, threads>>>(d_diff, d_thresh, size, 5); erosion<<<b2, t2>>>(d_thresh, d_morph, width, height); cudaEventRecord(stop); cudaEventSynchronize(stop); float ms; cudaEventElapsedTime(&ms, start, stop); // -------- COPY BACK -------- unsigned char* h_out = (unsigned char*)malloc(size); cudaMemcpy(h_out, d_diff, size, cudaMemcpyDeviceToHost); // -------- SAVE -------- stbi_write_jpg(out_name, width, height, 1, h_out, 100); printf("Frame %d processed | Time: %.3f ms | Saved: %s\n", i, ms, out_name); int motion_pixels = 0; for (int i = 0; i < size; i++) { if (h_out[i] == 255) motion_pixels++; } float percent = (motion_pixels * 100.0f) / size; if (percent > 1.0) { printf("Frame %d: Motion detected (%.2f%%)\n", i, percent); } // cleanup stbi_image_free(h_prev); stbi_image_free(h_curr); free(h_out); cudaFree(d_prev); cudaFree(d_curr); cudaFree(d_diff); cudaFree(d_thresh); cudaFree(d_morph); } return 0; }
